Программный код генерации на C++
int main() {
    const int SEQUENCE_LENGTH = 128;
    ofstream file("sequence_cpp.txt");
    random_device rd;
    mt19937 gen(rd());
    uniform_int_distribution<> dist(0, 1);
    for (int i = 0; i < SEQUENCE_LENGTH; i++) {
        file << dist(gen);
    }
    file.close();
    return 0;
}

Программный код генерации на Java:
public class Generator {
    public static void main(String[] args) {
        final int SEQUENCE_LENGTH = 128;
        try (PrintWriter writer = new PrintWriter(new FileWriter("sequence_java.txt"))) {
            Random random = new Random();
            for (int i = 0; i < SEQUENCE_LENGTH; i++) {
                writer.print(random.nextInt(2));
            }
        } catch (Exception e) {}
    }
}


2. РЕЗУЛЬТАТЫ ТЕСТИРОВАНИЯ
Программный код тестов NIST на Python:
def erfc(x): return math.erfc(x)
def igamc(a,x):
    if a==1.5:
        s=math.sqrt(x)
        return 2*erfc(s)+2*s*math.exp(-x)/math.sqrt(math.pi)
    return 0
def freq_test(seq):
    n=len(seq)
    s=sum([1 if b==1 else -1 for b in seq])/math.sqrt(n)
    return erfc(abs(s)/math.sqrt(2))
def runs_test(seq):
    n=len(seq)
    p=sum(seq)/n
    if abs(p-0.5)>=2/math.sqrt(n): return 0
    v=0
    for i in range(n-1):
        if seq[i]!=seq[i+1]: v+=1
    num=abs(v-2*n*p*(1-p))
    den=2*math.sqrt(2*n)*p*(1-p)
    return erfc(num/den) if den!=0 else 0
def long_test(seq):
    if len(seq)<128: return 0
    seq=seq[:128]
    v=[0,0,0,0]
    pi=[0.2148,0.3672,0.2305,0.1875]
    for i in range(16):
        b=seq[i*8:(i+1)*8]
        m=c=0
        for bit in b:
            if bit==1:
                c+=1
                m=max(m,c)
            else: c=0
        if m<=1: v[0]+=1
        elif m==2: v[1]+=1
        elif m==3: v[2]+=1
        else: v[3]+=1
    chi=sum([(v[i]-16*pi[i])**2/(16*pi[i]) for i in range(4)])
    return igamc(1.5,chi/2)

3. ВЫВОДЫ

1. Обе сгенерированные последовательности успешно прошли все три теста NIST, так как полученные P-value превышают пороговое значение 0.01.

2. C++ mt19937 показал более сбалансированное соотношение нулей и единиц (49.2%/50.8%), Java Random имеет небольшое смещение в сторону единиц (47.7%/52.3%).
